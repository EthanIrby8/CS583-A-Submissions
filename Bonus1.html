class Worker:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.s = x.shape[0]
        self.d = x.shape[1]
        self.w = np.zeros((d, 1))
        
    def set_param(self, w):
        self.w = w
        
    def loss(self):
        yx = np.multiply(self.y, self.x)
        yxw = np.dot(yx, self.w)
        vec1 = np.exp(-yxw)
        vec2 = np.log(1 + vec1)
        return np.sum(vec2)
    
    def sgd_loss(self):
        yx = np.multiply(self.y, self.x)
        yxw = np.dot(yx, self.w)
        vec1 = np.exp(-yxw)
        loss = np.log(1 + vec1)
        return loss
    
    def gradient(self, lam, w):
        yx = np.multiply(self.y, self.x)
        yxw = np.dot(yx, w)
        vec1 = np.exp(yxw)
        vec2 = np.divide(yx, 1 + vec1)
        vec3 = -np.sum(vec2, axis=0).reshape(self.d, 1) 
        g = (1/m) * (vec3 + lam * w)
        return g
    
    def local_update_gd(self, lam, alpha, w=None):
        if w is None:
            self.w = w
        
        for t in range(max_epoch):
            worker = Worker(self.x,self.y)
            g = worker.gradient(lam, w)
            self.w -= alpha * g
            wtildai = self.w
        alpha *= 0.9
        
        return wtildai
    
    def stochastic_gradient(self, lam, w):
        yx = self.x.T * self.y
        yxw = float(np.dot(yx.T,w)) 
        sg_loss = -np.sum(yx.T / (1 + np.exp(yxw)), axis=0).reshape(self.d, 1) #1x9 / 1x1
        sg = sg_loss + lam * w 
        return sg 

    def local_update_sgd(self, alpha, w=None):
        if w is None:
            self.w = w
        
        for t in range(max_epoch):
            rand_indices = np.random.permutation(self.s)
            x_rand = self.x[rand_indices, :]
            y_rand = self.y[rand_indices, :]
    
            for sample in range(self.s):
                xi = x_rand[sample, :]
                yi = float(y_rand[sample, :])
                worker = Worker(xi, yi)
                sg = worker.stochastic_gradient(lam,w)
                self.w -= alpha * sg
                wtildai = self.w
            alpha *= 0.9
        return wtildai

class Server:
    def __init__(self, m, n, d):
        self.m = m
        self.n = n
        self.d = d
        self.w = np.zeros((d, 1))
        self.obj = 0
        self.loss = 0
        
    def broadcast(self):
        return self.w    
            
    def aggregate(self, wtildam, losses):
        # wtildam: a list of d-by-1 vectors
        # losses: a list of scalars
        self.w = np.zeros((self.d, 1))
        self.loss = 0
        for i in range(self.m):
            self.w += wtildam[i]
            self.loss += losses[i]
            
    def objective(self, lam):
        reg = lam / 2 * np.sum(self.w * self.w)
        self.obj = self.loss / self.n + reg
